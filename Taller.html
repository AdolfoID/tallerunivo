📌 Fundamentos que debes dominar

Antes de programar, revisa estos conceptos. En clase los aplicarás construyendo soluciones reales con tu equipo.

🏷️

Clase

Una clase es una plantilla o molde que define atributos y comportamientos comunes a un grupo de objetos.

🎯

Objeto

Un objeto es una instancia concreta de una clase, con sus propios valores de atributos.

⚙️

Atributos

Son los datos que describen el estado de un objeto (nombre, edad, saldo, etc.).

🔧

Métodos

Son las acciones que puede realizar un objeto (saludar, calcular, mostrar información).

class Estudiante {
  constructor(nombre, edad, carrera) {
    this.nombre        = nombre;
    this.edad          = edad;
    this.carrera       = carrera;
    this.calificaciones = [];
  }

  agregarNota(nota) {
    if (nota >= 0 && nota <= 10) {
      this.calificaciones.push(nota);
      return "✅ Nota agregada";
    }
    return "❌ Nota inválida (0–10)";
  }

  obtenerPromedio() {
    if (this.calificaciones.length === 0) return 0;
    const sum = this.calificaciones.reduce((a, b) => a + b, 0);
    return (sum / this.calificaciones.length).toFixed(2);
  }

  mostrarInfo() {
    return `${this.nombre} (${this.edad} años) — ${this.carrera} — Promedio: ${this.obtenerPromedio()}`;
  }
}

  • Método precioFinal() que devuelva el precio con descuento aplicado

class Producto {
  constructor(nombre, precio, stock, descuento = 0) {
    this.nombre    = nombre;
    this.precio    = precio;
    this.stock     = stock;
    this.descuento = descuento; // porcentaje 0–100
  }

  precioFinal() {
    const ahorro = this.precio * (this.descuento / 100);
    return this.precio - ahorro;
  }

  vender(cantidad) {
    if (cantidad <= this.stock) {
      this.stock -= cantidad;
      return `✅ Venta OK. Stock restante: ${this.stock}`;
    }
    return `❌ Stock insuficiente. Solo quedan ${this.stock}`;
  }

  mostrarInfo() {
    return `📦 ${this.nombre} | S/ ${this.precio} | -${this.descuento}% | Final: S/ ${this.precioFinal().toFixed(2)} | Stock: ${this.stock}`;
  }
}

class Personaje {
  constructor(nombre, vida, fuerza) {
    this.nombre = nombre;
    this.vida   = vida;
    this.fuerza = fuerza;
    this.nivel  = 1;
  }

  atacar(enemigo) {
    const daño = this.fuerza * this.nivel;
    enemigo.vida -= daño;
    return `⚔️ ${this.nombre} causa ${daño} de daño. ${enemigo.nombre} queda con ${enemigo.vida} HP`;
  }

  subirNivel() {
    this.nivel++;
    this.fuerza += 5;
    return `🆙 ${this.nombre} → Nivel ${this.nivel} | Fuerza: ${this.fuerza}`;
  }

  curar() {
    this.vida += 20;
    return `💚 ${this.nombre} se cura → HP: ${this.vida}`;
  }

  estado() {
    return `⚔️ ${this.nombre} | Nv.${this.nivel} | ❤️ ${this.vida} HP | 💪 ${this.fuerza} ATK`;
  }
}

class MaterialBiblioteca {
  constructor(id, titulo, año) {
    this.id         = id;
    this.titulo     = titulo;
    this.año        = año;
    this.disponible = true;
  }
  prestar() {
    if (this.disponible) {
      this.disponible = false;
      return `📖 "${this.titulo}" prestado.`;
    }
    return `❌ "${this.titulo}" no disponible.`;
  }
  devolver() {
    this.disponible = true;
    return `📚 "${this.titulo}" devuelto.`;
  }
  info() {
    return `[${this.id}] ${this.titulo} (${this.año}) — ${this.disponible ? "Disponible" : "Prestado"}`;
  }
}

// ── Clase hija: Libro ──────────────────────────
class Libro extends MaterialBiblioteca {
  constructor(id, titulo, año, autor, genero) {
    super(id, titulo, año);
    this.autor  = autor;
    this.genero = genero;
  }
  infoCompleta() {
    return `📘 ${super.info()} | Autor: ${this.autor} | Género: ${this.genero}`;
  }
}

// ── Clase hija: Revista ────────────────────────
class Revista extends MaterialBiblioteca {
  constructor(id, titulo, año, edicion, mes) {
    super(id, titulo, año);
    this.edicion = edicion;
    this.mes     = mes;
  }
  infoCompleta() {
    return `📰 ${super.info()} | Ed. ${this.edicion} | Mes: ${this.mes}`;
  }
}
